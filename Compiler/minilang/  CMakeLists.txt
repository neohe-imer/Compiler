cmake_minimum_required(VERSION 3.16)
project(minilang CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable(minilang
  src/main.cpp
  src/lexer.cpp
  src/parser.cpp
  src/interpreter.cpp
  src/compiler.cpp
  src/vm.cpp
)

target_include_directories(minilang PRIVATE src)
